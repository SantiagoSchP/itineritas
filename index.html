// index.ts // Edge Function: send-daily-summary // // Que hace: revisa todos los usuarios guardados en la tabla user_data, // y para cada uno que tenga activado el resumen diario y cuya hora // configurada ya haya llegado (segun su zona horaria), le manda un // correo con las actividades del cronograma de hoy. Usa la tabla // daily_summary_log para no mandar el mismo correo dos veces el mismo dia. // // Esta funcion esta pensada para ser llamada automaticamente cada // 15 minutos por un "cron job" (programador de tareas) de Supabase. import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'; const SUPABASE_URL = Deno.env.get('SUPABASE_URL'); const SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY'); const RESEND_API_KEY = Deno.env.get('RESEND_API_KEY'); const supabase = createClient(SUPABASE_URL, SERVICE_ROLE_KEY); // Convierte el cronograma de un dia a un texto simple para el email function armarTextoResumen(cronogramaHoy, nombre) { if (!cronogramaHoy || cronogramaHoy.length === 0) { return `Hola ${nombre || ''}!\n\nNo tenes actividades cargadas en el cronograma para hoy.\n\n— Itineritas`; } const ordenado = cronogramaHoy.slice().sort((a, b) => (a.hora || '').localeCompare(b.hora || '')); const lineas = ordenado.map((c) => { const lugar = c.lugar ? ` (${c.lugar})` : ''; return `${c.hora || '--:--'} — ${c.desc || ''}${lugar}`; }); return `Hola ${nombre || ''}!\n\nEste es el cronograma de hoy:\n\n${lineas.join('\n')}\n\n— Itineritas`; } // Envia un email usando la API de Resend async function enviarEmail(destinatario, asunto, textoPlano) { const resp = await fetch('https://api.resend.com/emails', { method: 'POST', headers: { 'Authorization': `Bearer ${RESEND_API_KEY}`, 'Content-Type': 'application/json', }, body: JSON.stringify({ from: 'Itineritas ', to: [destinatario], subject: asunto, text: textoPlano, }), }); if (!resp.ok) { const errorBody = await resp.text(); throw new Error(`Resend respondio ${resp.status}: ${errorBody}`); } return resp.json(); } // Corrige variantes comunes de nombres de zona horaria que algunos // dispositivos entregan de forma incompleta o no estandar. function normalizarZona(zona) { if (!zona) return 'America/Argentina/Buenos_Aires'; const mapa = { 'America/Buenos_Aires': 'America/Argentina/Buenos_Aires', 'America/Cordoba': 'America/Argentina/Cordoba', 'America/Mendoza': 'America/Argentina/Mendoza', 'America/Catamarca': 'America/Argentina/Catamarca', 'America/Jujuy': 'America/Argentina/Jujuy', 'America/Salta': 'America/Argentina/Salta', 'America/Tucuman': 'America/Argentina/Tucuman', }; if (mapa[zona]) return mapa[zona]; // Verificar si la zona es valida probandola directamente; // si Intl la rechaza, usamos Buenos Aires como respaldo razonable. try { new Intl.DateTimeFormat('en-GB', { timeZone: zona }); return zona; // es valida tal cual } catch (e) { console.error(`Zona horaria no reconocida: "${zona}". Usando America/Argentina/Buenos_Aires como respaldo.`); return 'America/Argentina/Buenos_Aires'; } } // Calcula la hora actual (HH:MM) en una zona horaria especifica function horaActualEnZona(zona) { try { const ahora = new Date(); const formatter = new Intl.DateTimeFormat('en-GB', { timeZone: zona, hour: '2-digit', minute: '2-digit', hour12: false, }); return formatter.format(ahora); // "HH:MM" } catch (e) { console.error(`Error calculando hora en zona "${zona}": ${e}`); const ahora = new Date(); return `${String(ahora.getUTCHours()).padStart(2, '0')}:${String(ahora.getUTCMinutes()).padStart(2, '0')}`; } } function fechaHoyEnZona(zona) { try { const ahora = new Date(); const formatter = new Intl.DateTimeFormat('en-CA', { timeZone: zona }); // formato YYYY-MM-DD return formatter.format(ahora); } catch (e) { return new Date().toISOString().slice(0, 10); } } Deno.serve(async (req) => { try { // Traer todos los usuarios const { data: usuarios, error } = await supabase .from('user_data') .select('email, name, data'); if (error) throw error; const resultados = []; for (const usuario of usuarios || []) { const config = usuario.data?.resumenDiario; // Si el usuario no activo esta funcion, lo saltamos if (!config || !config.activado || !config.hora) continue; const zona = normalizarZona(config.zona); const horaActual = horaActualEnZona(zona); const fechaHoy = fechaHoyEnZona(zona); // Solo mandamos si la hora configurada ya llego (con margen de 15 min, // ya que esta funcion se ejecuta cada 15 minutos) const [hConfig, mConfig] = config.hora.split(':').map(Number); const [hActual, mActual] = horaActual.split(':').map(Number); const minutosConfig = hConfig * 60 + mConfig; const minutosActual = hActual * 60 + mActual; const yaLlegoLaHora = minutosActual >= minutosConfig && (minutosActual - minutosConfig) < 15; console.log(`Usuario ${usuario.email}: zona=${zona}, horaActual=${horaActual}, horaConfig=${config.hora}, yaLlegoLaHora=${yaLlegoLaHora}`); if (!yaLlegoLaHora) continue; // Revisar si ya se le mando hoy const { data: logRow } = await supabase .from('daily_summary_log') .select('last_sent_date') .eq('email', usuario.email) .maybeSingle(); if (logRow && logRow.last_sent_date === fechaHoy) continue; // ya se mando hoy // Armar el cronograma de hoy const cronograma = usuario.data?.cronograma || []; const cronogramaHoy = cronograma.filter((c) => c.fecha === fechaHoy); const destino = config.emailDestino || usuario.email; const texto = armarTextoResumen(cronogramaHoy, usuario.name); await enviarEmail(destino, 'Tu resumen de hoy — Itineritas', texto); // Actualizar el registro de control await supabase .from('daily_summary_log') .upsert({ email: usuario.email, last_sent_date: fechaHoy }); resultados.push({ email: usuario.email, enviado: true }); } return new Response(JSON.stringify({ ok: true, resultados }), { headers: { 'Content-Type': 'application/json' }, }); } catch (err) { return new Response(JSON.stringify({ ok: false, error: String(err) }), { status: 500, headers: { 'Content-Type': 'application/json' }, });
